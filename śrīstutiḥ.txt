॥ 𑌶𑍍𑌰𑍀𑌸𑍍𑌤𑍁𑌤𑌿𑌃 ॥

𑌶𑍍𑌰𑍀𑌮𑌾𑌨𑍍𑌵𑍇𑌙𑍍𑌕𑌟𑌨𑌾𑌥𑌾𑌰𑍍𑌯𑌃 𑌕𑌵𑌿𑌤𑌾𑌰𑍍𑌕𑌿𑌕𑌕𑍇𑌸𑌰𑍀 ।
𑌵𑍇𑌦𑌾𑌨𑍍𑌤𑌾𑌚𑌾𑌰𑍍𑌯𑌵𑌰𑍍𑌯𑍋 𑌮𑍇 𑌸𑌨𑍍𑌨𑌿𑌧𑌤𑍍𑌤𑌾𑌂 𑌸𑌦𑌾 𑌹𑍃𑌦𑌿 ॥

𑌮𑌾𑌨𑌾𑌤𑍀𑌤𑌪𑍍𑌰𑌥𑌿𑌤𑌵𑌿𑌭𑌵𑌾𑌂 𑌮𑌙𑍍𑌗𑌲𑌂 𑌮𑌙𑍍𑌗𑌲𑌾𑌨𑌾𑌂 𑌵𑌕𑍍𑌷𑌃𑌪𑍀𑌠𑍀𑌂 𑌮𑌧𑍁𑌵𑌿𑌜𑌯𑌿𑌨𑍋 𑌭𑍂𑌷𑌯𑌨𑍍𑌤𑍀𑌂 𑌸𑍍𑌵𑌕𑌾𑌨𑍍𑌤𑍍𑌯𑌾 ।
𑌪𑍍𑌰𑌤𑍍𑌯𑌕𑍍𑌷𑌾𑌨𑍁𑌶𑍍𑌰𑌵𑌿𑌕𑌮𑌹𑌿𑌮𑌪𑍍𑌰𑌾𑌰𑍍𑌥𑌿𑌨𑍀𑌨𑌾𑌂 𑌪𑍍𑌰𑌜𑌾𑌨𑌾𑌂 𑌶𑍍𑌰𑍇𑌯𑍋𑌮𑍂𑌰𑍍𑌤𑌿𑌂 𑌶𑍍𑌰𑌿𑌯𑌮𑌶𑌰𑌣𑌸𑍍𑌤𑍍𑌵𑌾𑌂 𑌶𑌰𑌣𑍍𑌯𑌾𑌂 𑌪𑍍𑌰𑌪𑌦𑍍𑌯𑍇 ॥௧॥

𑌆𑌵𑌿𑌰𑍍𑌭𑌾𑌵𑌃 𑌕𑌲𑌶𑌜𑌲𑌧𑌾𑌵𑌧𑍍𑌵𑌰𑍇 𑌵𑌾𑌪𑌿 𑌯𑌸𑍍𑌯𑌾𑌃 𑌸𑍍𑌥𑌾𑌨𑌂 𑌯𑌸𑍍𑌯𑌾𑌃 𑌸𑌰𑌸𑌿𑌜𑌵𑌨𑌂 𑌵𑌿𑌷𑍍𑌣𑍁𑌵𑌕𑍍𑌷𑌃𑌸𑍍𑌥𑌲𑌂 𑌵𑌾 ।
𑌭𑍂𑌮𑌾 𑌯𑌸𑍍𑌯𑌾 𑌭𑍁𑌵𑌨𑌮𑌖𑌿𑌲𑌂 𑌦𑍇𑌵𑌿 𑌦𑌿𑌵𑍍𑌯𑌂 𑌪𑌦𑌂 𑌵𑌾 𑌸𑍍𑌤𑍋𑌕𑌪𑍍𑌰𑌜𑍍𑌞𑍈𑌰𑌨𑌵𑌧𑌿𑌗𑍁𑌣𑌾 𑌸𑍍𑌤𑍂𑌯𑌸𑍇 𑌸𑌾 𑌕𑌥𑌂 𑌤𑍍𑌵𑌮𑍍 ॥௨॥

𑌸𑍍𑌤𑍋𑌤𑌵𑍍𑌯𑌤𑍍𑌵𑌂 𑌦𑌿𑌶𑌤𑌿 𑌭𑌵𑌤𑍀 𑌦𑍇𑌹𑌿𑌭𑌿𑌃 𑌸𑍍𑌤𑍂𑌯𑌮𑌾𑌨𑌾 𑌤𑌾𑌮𑍇𑌵 𑌤𑍍𑌵𑌾𑌮𑌨𑌿𑌤𑌰𑌗𑌤𑌿𑌃 𑌸𑍍𑌤𑍋𑌤𑍁𑌮𑌾𑌶𑌂𑌸𑌮𑌾𑌨𑌃 ।
𑌸𑌿𑌦𑍍𑌧𑌾𑌰𑌮𑍍𑌭𑌃 𑌸𑌕𑌲𑌭𑍁𑌵𑌨𑌶𑍍𑌲𑌾𑌘𑌨𑍀𑌯𑍋 𑌭𑌵𑍇𑌯𑌂 𑌸𑍇𑌵𑌾𑌪𑍇𑌕𑍍𑌷𑌾 𑌤𑌵 𑌚𑌰𑌣𑌯𑍋𑌃 𑌶𑍍𑌰𑍇𑌯𑌸𑍇 𑌕𑌸𑍍𑌯 𑌨 𑌸𑍍𑌯𑌾𑌤𑍍 ॥௩॥

𑌯𑌤𑍍𑌸𑌙𑍍𑌕𑌲𑍍𑌪𑌾𑌦𑍍𑌭𑌵𑌤𑌿 𑌕𑌮𑌲𑍇 𑌯𑌤𑍍𑌰 𑌦𑍇𑌹𑌿𑌨𑍍𑌯𑌮𑍀𑌷𑌾𑌂 𑌜𑌨𑍍𑌮𑌸𑍍𑌥𑍇𑌮𑌪𑍍𑌰𑌲𑌯𑌰𑌚𑌨𑌾 𑌜𑌙𑍍𑌗𑌮𑌾𑌜𑌙𑍍𑌗𑌮𑌾𑌨𑌾𑌮𑍍 ।
𑌤𑌤𑍍𑌕𑌲𑍍𑌯𑌾𑌣𑌂 𑌕𑌿𑌮𑌪𑌿 𑌯𑌮𑌿𑌨𑌾𑌮𑍇𑌕𑌲𑌕𑍍𑌷𑍍𑌯𑌂 𑌸𑌮𑌾𑌧𑍌 𑌪𑍂𑌰𑍍𑌣𑌂 𑌤𑍇𑌜𑌃 𑌸𑍍𑌫𑍁𑌰𑌤𑌿 𑌭𑌵𑌤𑍀𑌪𑌾𑌦𑌲𑌾𑌕𑍍𑌷𑌾𑌰𑌸𑌾𑌙𑍍𑌕𑌮𑍍 ॥௪॥

𑌨𑌿𑌷𑍍𑌪𑍍𑌰𑌤𑍍𑌯𑍂𑌹𑌪𑍍𑌰𑌣𑌯𑌘𑌟𑌿𑌤𑌂 𑌦𑍇𑌵𑌿 𑌨𑌿𑌤𑍍𑌯𑌾𑌨𑌪𑌾𑌯𑌂 𑌵𑌿𑌷𑍍𑌣𑍁𑌸𑍍𑌤𑍍𑌵𑌂 𑌚𑍇𑌤𑍍𑌯𑌨𑌵𑌧𑌿𑌗𑍁𑌣𑌂 𑌦𑍍𑌵𑌨𑍍𑌦𑍍𑌵𑌮𑌨𑍍𑌯𑍋𑌨𑍍𑌯𑌲𑌕𑍍𑌷𑍍𑌯𑌮𑍍 ।
𑌶𑍇𑌷𑌶𑍍𑌚𑌿𑌤𑍍𑌤𑌂 𑌵𑌿𑌮𑌲𑌮𑌨𑌸𑌾𑌂 𑌮𑍌𑌲𑌯𑌶𑍍𑌚 𑌶𑍍𑌰𑍁𑌤𑍀𑌨𑌾𑌂 𑌸𑌮𑍍𑌪𑌦𑍍𑌯𑌨𑍍𑌤𑍇 𑌵𑌿𑌹𑌰𑌣𑌵𑌿𑌧𑍌 𑌯𑌸𑍍𑌯 𑌶𑌯𑍍𑌯𑌾𑌵𑌿𑌶𑍇𑌷𑌾𑌃 ॥௫॥

𑌉𑌦𑍍𑌦𑍇𑌶𑍍𑌯𑌤𑍍𑌵𑌂 𑌜𑌨𑌨𑌿 𑌭𑌜𑌤𑍋𑌰𑍂𑌜𑍍𑌝𑌿𑌤𑍋𑌪𑌾𑌧𑌿𑌗𑌨𑍍𑌧𑌂 𑌪𑍍𑌰𑌤𑍍𑌯𑌗𑍍𑌰𑍂𑌪𑍇 𑌹𑌵𑌿𑌷𑌿 𑌯𑍁𑌵𑌯𑍋𑌰𑍇𑌕𑌶𑍇𑌷𑌿𑌤𑍍𑌵𑌯𑍋𑌗𑌾𑌤𑍍 ।
𑌪𑌦𑍍𑌮𑍇 𑌪𑌤𑍍𑌯𑍁𑌸𑍍𑌤𑌵 𑌚 𑌨𑌿𑌗𑌮𑍈𑌰𑍍𑌨𑌿𑌤𑍍𑌯𑌮𑌨𑍍𑌵𑌿𑌷𑍍𑌯𑌮𑌾𑌣𑍋 𑌨𑌾𑌵𑌚𑍍𑌛𑍇𑌦𑌂 𑌭𑌜𑌤𑌿 𑌮𑌹𑌿𑌮𑌾 𑌨𑌰𑍍𑌤𑌯𑌨𑍍𑌮𑌾𑌨𑌸𑌂 𑌨𑌃 ॥௬॥

𑌪𑌶𑍍𑌯𑌨𑍍𑌤𑍀𑌷𑍁 𑌶𑍍𑌰𑍁𑌤𑌿𑌷𑍁 𑌪𑌰𑌿𑌤𑌃 𑌸𑍂𑌰𑌿𑌬𑍃𑌨𑍍𑌦𑍇𑌨 𑌸𑌾𑌰𑍍𑌧𑌂 𑌮𑌧𑍍𑌯𑍇𑌕𑍃𑌤𑍍𑌯 𑌤𑍍𑌰𑌿𑌗𑍁𑌣𑌫𑌲𑌕𑌂 𑌨𑌿𑌰𑍍𑌮𑌿𑌤𑌸𑍍𑌥𑌾𑌨𑌭𑍇𑌦𑌮𑍍 ।
𑌵𑌿𑌶𑍍𑌵𑌾𑌧𑍀𑌶𑌪𑍍𑌰𑌣𑌯𑌿𑌨𑌿 𑌸𑌦𑌾 𑌵𑌿𑌭𑍍𑌰𑌮𑌦𑍍𑌯𑍂𑌤𑌵𑍃𑌤𑍍𑌤𑍌 𑌬𑍍𑌰𑌹𑍍𑌮𑍇𑌶𑌾𑌦𑍍𑌯𑌾 𑌦𑌧𑌤𑌿 𑌯𑍁𑌵𑌯𑍋𑌰𑌕𑍍𑌷𑌶𑌾𑌰𑌪𑍍𑌰𑌚𑌾𑌰𑌮𑍍 ॥௭॥

𑌅𑌸𑍍𑌯𑍇𑌶𑌾𑌨𑌾 𑌤𑍍𑌵𑌮𑌸𑌿 𑌜𑌗𑌤𑌃 𑌸𑌂𑌶𑍍𑌰𑌯𑌨𑍍𑌤𑍀 𑌮𑍁𑌕𑍁𑌨𑍍𑌦𑌂 𑌲𑌕𑍍𑌷𑍍𑌮𑍀𑌃 𑌪𑌦𑍍𑌮𑌾 𑌜𑌲𑌧𑌿𑌤𑌨𑌯𑌾 𑌵𑌿𑌷𑍍𑌣𑍁𑌪𑌤𑍍𑌨𑍀𑌨𑍍𑌦𑌿𑌰𑍇𑌤𑌿 ।
𑌯𑌨𑍍𑌨𑌾𑌮𑌾𑌨𑌿 𑌶𑍍𑌰𑍁𑌤𑌿𑌪𑌰𑌿𑌪𑌣𑌾𑌨𑍍𑌯𑍇𑌵𑌮𑌾𑌵𑌰𑍍𑌤𑌯𑌨𑍍𑌤𑍋 𑌨𑌾𑌵𑌰𑍍𑌤𑌨𑍍𑌤𑍇 𑌦𑍁𑌰𑌿𑌤𑌪𑌵𑌨𑌪𑍍𑌰𑍇𑌰𑌿𑌤𑍇 𑌜𑌨𑍍𑌮𑌚𑌕𑍍𑌰𑍇 ॥௮॥

𑌤𑍍𑌵𑌾𑌮𑍇𑌵𑌾𑌹𑍁𑌃 𑌕𑌤𑌿𑌚𑌿𑌦𑌪𑌰𑍇 𑌤𑍍𑌵𑌤𑍍𑌪𑍍𑌰𑌿𑌯𑌂 𑌲𑍋𑌕𑌨𑌾𑌥𑌂 𑌕𑌿𑌂 𑌤𑍈𑌰𑌨𑍍𑌤𑌃𑌕𑌲𑌹𑌮𑌲𑌿𑌨𑍈𑌃 𑌕𑌿𑌞𑍍𑌚𑌿𑌦𑍁𑌤𑍍𑌤𑍀𑌰𑍍𑌯 𑌮𑌗𑍍𑌨𑍈𑌃 ।
𑌤𑍍𑌵𑌤𑍍𑌸𑌮𑍍𑌪𑍍𑌰𑍀𑌤𑍍𑌯𑍈 𑌵𑌿𑌹𑌰𑌤𑌿 𑌹𑌰𑍌 𑌸𑌮𑍍𑌮𑍁𑌖𑍀𑌨𑌾𑌂 𑌶𑍍𑌰𑍁𑌤𑍀𑌨𑌾𑌂 𑌭𑌾𑌵𑌾𑌰𑍂𑌢𑍌 𑌭𑌗𑌵𑌤𑌿 𑌯𑍁𑌵𑌾𑌂 𑌦𑌮𑍍𑌪𑌤𑍀 𑌦𑍈𑌵𑌤𑌂 𑌨𑌃 ॥௯॥

𑌆𑌪𑌨𑍍𑌨𑌾𑌰𑍍𑌤𑌿𑌪𑍍𑌰𑌶𑌮𑌨𑌵𑌿𑌧𑍌 𑌬𑌦𑍍𑌧𑌦𑍀𑌕𑍍𑌷𑌸𑍍𑌯 𑌵𑌿𑌷𑍍𑌣𑍋𑌰𑌾𑌚𑌖𑍍𑌯𑍁𑌸𑍍𑌤𑍍𑌵𑌾𑌂 𑌪𑍍𑌰𑌿𑌯𑌸𑌹𑌚𑌰𑍀𑌮𑍈𑌕𑌮𑌤𑍍𑌯𑍋𑌪𑌪𑌨𑍍𑌨𑌾𑌮𑍍 ।
𑌪𑍍𑌰𑌾𑌦𑍁𑌰𑍍𑌭𑌾𑌵𑍈𑌰𑌪𑌿 𑌸𑌮𑌤𑌨𑍁𑌃 𑌪𑍍𑌰𑌾𑌧𑍍𑌵𑌮𑌨𑍍𑌵𑍀𑌯𑌸𑍇 𑌤𑍍𑌵𑌂 𑌦𑍂𑌰𑍋𑌤𑍍𑌕𑍍𑌷𑌿𑌪𑍍𑌤𑍈𑌰𑌿𑌵 𑌮𑌧𑍁𑌰𑌤𑌾 𑌦𑍁𑌗𑍍𑌧𑌰𑌾𑌶𑍇𑌸𑍍𑌤𑌰𑌙𑍍𑌗𑍈𑌃 ॥௰॥

𑌧𑌤𑍍𑌤𑍇 𑌶𑍋𑌭𑌾𑌂 𑌹𑌰𑌿𑌮𑌰𑌕𑌤𑍇 𑌤𑌾𑌵𑌕𑍀 𑌮𑍂𑌰𑍍𑌤𑌿𑌰𑌾𑌦𑍍𑌯𑌾 𑌤𑌨𑍍𑌵𑍀 𑌤𑍁𑌙𑍍𑌗𑌸𑍍𑌤𑌨𑌭𑌰𑌨𑌤𑌾 𑌤𑌪𑍍𑌤𑌜𑌾𑌮𑍍𑌬𑍂𑌨𑌦𑌾𑌭𑌾 ।
𑌯𑌸𑍍𑌯𑌾𑌂 𑌗𑌚𑍍𑌛𑌨𑍍𑌤𑍍𑌯𑍁𑌦𑌯𑌵𑌿𑌲𑌯𑍈𑌰𑍍𑌨𑌿𑌤𑍍𑌯𑌮𑌾𑌨𑌨𑍍𑌦𑌸𑌿𑌨𑍍𑌧𑌾𑌵𑌿𑌚𑍍𑌛𑌾𑌵𑍇𑌗𑍋𑌲𑍍𑌲𑌸𑌿𑌤𑌲𑌹𑌰𑍀𑌵𑌿𑌭𑍍𑌰𑌮𑌂 𑌵𑍍𑌯𑌕𑍍𑌤𑌯𑌸𑍍𑌤𑍇 ॥௰௧॥

𑌆𑌸𑌂𑌸𑌾𑌰𑌂 𑌵𑌿𑌤𑌤𑌮𑌖𑌿𑌲𑌂 𑌵𑌾𑌙𑍍𑌮𑌯𑌂 𑌯𑌦𑍍𑌵𑌿𑌭𑍂𑌤𑌿𑌰𑍍𑌯𑌦𑍍𑌭𑍍𑌰𑍂𑌭𑌙𑍍𑌗𑌾𑌤𑍍𑌕𑍁𑌸𑍁𑌮𑌧𑌨𑍁𑌷𑌃 𑌕𑌿𑌙𑍍𑌕𑌰𑍋 𑌮𑍇𑌰𑍂𑌧𑌨𑍍𑌵𑌾 ।
𑌯𑌸𑍍𑌯𑌾𑌂 𑌨𑌿𑌤𑍍𑌯𑌂 𑌨𑌯𑌨𑌶𑌤𑌕𑍈𑌰𑍇𑌕𑌲𑌕𑍍𑌷𑍍𑌯𑍋 𑌮𑌹𑍇𑌨𑍍𑌦𑍍𑌰𑌃 𑌪𑌦𑍍𑌮𑍇 𑌤𑌾𑌸𑌾𑌂 𑌪𑌰𑌿𑌣𑌤𑌿𑌰𑌸𑍌 𑌭𑌾𑌵𑌲𑍇𑌶𑍈𑌸𑍍𑌤𑍍𑌵𑌦𑍀𑌯𑍈𑌃 ॥௰௨॥

𑌅𑌗𑍍𑌰𑍇 𑌭𑌰𑍍𑌤𑍁𑌃 𑌸𑌰𑌸𑌿𑌜𑌮𑌯𑍇 𑌭𑌦𑍍𑌰𑌪𑍀𑌠𑍇 𑌨𑌿𑌷𑌣𑍍𑌣𑌾𑌮𑌮𑍍𑌭𑍋𑌰𑌾𑌶𑍇𑌰𑌧𑌿𑌗𑌤𑌸𑍁𑌧𑌾𑌸𑌮𑍍𑌪𑍍𑌲𑌵𑌾𑌦𑍁𑌤𑍍𑌥𑌿𑌤𑌾𑌂 𑌤𑍍𑌵𑌾𑌮𑍍 ।
𑌪𑍁𑌷𑍍𑌪𑌾𑌸𑌾𑌰𑌸𑍍𑌥𑌗𑌿𑌤𑌭𑍁𑌵𑌨𑍈𑌃 𑌪𑍁𑌷𑍍𑌕𑌲𑌾𑌵𑌰𑍍𑌤𑌕𑌾𑌦𑍍𑌯𑍈𑌃 𑌕𑍢𑌪𑍍𑌤𑌾𑌰𑌮𑍍𑌭𑌾𑌃 𑌕𑌨𑌕𑌕𑌲𑌶𑍈𑌰𑌭𑍍𑌯𑌷𑌿𑌞𑍍𑌚𑌨𑍍𑌗𑌜𑍇𑌨𑍍𑌦𑍍𑌰𑌾𑌃 ॥௰௩॥

𑌆𑌲𑍋𑌕𑍍𑌯 𑌤𑍍𑌵𑌾𑌮𑌮𑍃𑌤𑌸𑌹𑌜𑍇 𑌵𑌿𑌷𑍍𑌣𑍁𑌵𑌕𑍍𑌷𑌃𑌸𑍍𑌥𑌲𑌸𑍍𑌥𑌾𑌂 𑌶𑌾𑌪𑌾𑌕𑍍𑌰𑌾𑌨𑍍𑌤𑌾𑌃 𑌶𑌰𑌣𑌮𑌗𑌮𑌨𑍍𑌸𑌾𑌵𑌰𑍋𑌧𑌾𑌃 𑌸𑍁𑌰𑍇𑌨𑍍𑌦𑍍𑌰𑌾𑌃 ।
𑌲𑌬𑍍𑌧𑍍𑌵𑌾 𑌭𑍂𑌯𑌸𑍍𑌤𑍍𑌰𑌿𑌭𑍁𑌵𑌨𑌮𑌿𑌦𑌂 𑌲𑌕𑍍𑌷𑌿𑌤𑌂 𑌤𑍍𑌵𑌤𑍍𑌕𑌟𑌾𑌕𑍍𑌷𑍈𑌃 𑌸𑌰𑍍𑌵𑌾𑌕𑌾𑌰𑌸𑍍𑌥𑌿𑌰𑌸𑌮𑍁𑌦𑌯𑌾𑌂 𑌸𑌮𑍍𑌪𑌦𑌂 𑌨𑌿𑌰𑍍𑌵𑌿𑌶𑌨𑍍𑌤𑌿 ॥௰௪॥

𑌆𑌰𑍍𑌤𑌤𑍍𑌰𑌾𑌣𑌤𑍍𑌰𑌵𑍍𑌰𑌤𑌿𑌭𑌿𑌰𑌮𑍃𑌤𑌾𑌸𑌾𑌰𑌨𑍀𑌲𑌾𑌮𑍍𑌬𑍁𑌵𑌾𑌹𑍈𑌰𑌮𑍍𑌭𑍋𑌜𑌾𑌨𑌾𑌮𑍁𑌷𑌸𑌿 𑌮𑌿𑌷𑌤𑌾𑌮𑌨𑍍𑌤𑌰𑌙𑍍𑌗𑍈𑌰𑌪𑌾𑌙𑍍𑌗𑍈𑌃 ।
𑌯𑌸𑍍𑌯𑌾𑌂 𑌯𑌸𑍍𑌯𑌾𑌂 𑌦𑌿𑌶𑌿 𑌵𑌿𑌹𑌰𑌤𑍇 𑌦𑍇𑌵𑌿 𑌦𑍃𑌷𑍍𑌟𑌿𑌸𑍍𑌤𑍍𑌵𑌦𑍀𑌯𑌾 𑌤𑌸𑍍𑌯𑌾𑌂 𑌤𑌸𑍍𑌯𑌾𑌮𑌹𑌮𑌹𑌮𑌿𑌕𑌾𑌂 𑌤𑌨𑍍𑌵𑌤𑍇 𑌸𑌮𑍍𑌪𑌦𑍋𑌘𑌾𑌃 ॥௰௫॥

𑌯𑍋𑌗𑌾𑌰𑌮𑍍𑌭𑌤𑍍𑌵𑌰𑌿𑌤𑌮𑌨𑌸𑍋 𑌯𑍁𑌷𑍍𑌮𑌦𑍈𑌕𑌾𑌨𑍍𑌤𑍍𑌯𑌯𑍁𑌕𑍍𑌤𑌂 𑌧𑌰𑍍𑌮𑌂 𑌪𑍍𑌰𑌾𑌪𑍍𑌤𑍁𑌂 𑌪𑍍𑌰𑌥𑌮𑌮𑌿𑌹 𑌯𑍇 𑌧𑌾𑌰𑌯𑌨𑍍𑌤𑍇𑌧𑌨𑌾 𑌯𑌾𑌮𑍍 ।
𑌤𑍇𑌷𑌾𑌂 𑌭𑍂𑌮𑍇𑌰𑍍𑌧𑌨𑌪𑌤𑌿𑌗𑍃𑌹𑌾𑌦𑌮𑍍𑌬𑌰𑌾𑌦𑌮𑍍𑌬𑍁𑌧𑍇𑌰𑍍𑌵𑌾 𑌧𑌾𑌰𑌾 𑌨𑌿𑌰𑍍𑌯𑌾𑌨𑍍𑌤𑍍𑌯𑌧𑌿𑌕𑌮𑌧𑌿𑌕𑌂 𑌵𑌾𑌞𑍍𑌛𑌿𑌤𑌾𑌨𑌾𑌂 𑌵𑌸𑍂𑌨𑌾𑌮𑍍 ॥௰௬॥

𑌶𑍍𑌰𑍇𑌯𑌸𑍍𑌕𑌾𑌮𑌾𑌃 𑌕𑌮𑌲𑌨𑌿𑌲𑌯𑍇 𑌚𑌿𑌤𑍍𑌰𑌮𑌾𑌮𑍍𑌨𑌾𑌯𑌵𑌾𑌚𑌾𑌂 𑌚𑍂𑌡𑌾𑌪𑍀𑌡𑌂 𑌤𑌵 𑌪𑌦𑌯𑍁𑌗𑌂 𑌚𑍇𑌤𑌸𑌾 𑌧𑌾𑌰𑌯𑌨𑍍𑌤𑌃 ।
𑌛𑌤𑍍𑌰𑌚𑍍𑌛𑌾𑌯𑌾𑌸𑍁𑌭𑌗𑌶𑌿𑌰𑌸𑌶𑍍𑌚𑌾𑌮𑌰𑌸𑍍𑌮𑍇𑌰𑌪𑌾𑌰𑍍𑌶𑍍𑌵𑌾𑌃 𑌶𑍍𑌲𑌾𑌘𑌾𑌶𑌬𑍍𑌦𑌶𑍍𑌰𑌵𑌣𑌮𑍁𑌦𑌿𑌤𑌾𑌃 𑌸𑍍𑌰𑌗𑍍𑌵𑌿𑌣𑌃 𑌸𑌞𑍍𑌚𑌰𑌨𑍍𑌤𑌿 ॥௰௭॥

𑌊𑌰𑍀𑌕𑌰𑍍𑌤𑍁𑌂 𑌕𑍁𑌶𑌲𑌮𑌖𑌿𑌲𑌂 𑌜𑍇𑌤𑍁𑌮𑌾𑌦𑍀𑌨𑌰𑌾𑌤𑍀𑌨𑍍𑌦𑍂𑌰𑍀𑌕𑌰𑍍𑌤𑍁𑌂 𑌦𑍁𑌰𑌿𑌤𑌨𑌿𑌵𑌹𑌂 𑌤𑍍𑌯𑌕𑍍𑌤𑍁𑌮𑌾𑌦𑍍𑌯𑌾𑌮𑌵𑌿𑌦𑍍𑌯𑌾𑌮𑍍 ।
𑌅𑌮𑍍𑌬 𑌸𑍍𑌤𑌮𑍍𑌬𑌾𑌵𑌧𑌿𑌕𑌜𑌨𑌨𑌗𑍍𑌰𑌾𑌮𑌸𑍀𑌮𑌾𑌨𑍍𑌤𑌰𑍇𑌖𑌾𑌮𑌾𑌲𑌮𑍍𑌬𑌨𑍍𑌤𑍇 𑌵𑌿𑌮𑌲𑌮𑌨𑌸𑍋 𑌵𑌿𑌷𑍍𑌣𑍁𑌕𑌾𑌨𑍍𑌤𑍇 𑌦𑌯𑌾𑌂 𑌤𑍇 ॥௰௮॥

𑌜𑌾𑌤𑌾𑌕𑌾𑌙𑍍𑌕𑍍𑌷𑌾 𑌜𑌨𑌨𑌿 𑌯𑍁𑌵𑌯𑍋𑌰𑍇𑌕𑌸𑍇𑌵𑌾𑌧𑌿𑌕𑌾𑌰𑍇 𑌮𑌾𑌯𑌾𑌲𑍀𑌢𑌂 𑌵𑌿𑌭𑌵𑌮𑌖𑌿𑌲𑌂 𑌮𑌨𑍍𑌯𑌮𑌾𑌨𑌾𑌸𑍍𑌤𑍃𑌣𑌾𑌯 ।
𑌪𑍍𑌰𑍀𑌤𑍍𑌯𑍈 𑌵𑌿𑌷𑍍𑌣𑍋𑌸𑍍𑌤𑌵 𑌚 𑌕𑍃𑌤𑌿𑌨𑌃 𑌪𑍍𑌰𑍀𑌤𑌿𑌮𑌨𑍍𑌤𑍋 𑌭𑌜𑌨𑍍𑌤𑍇 𑌵𑍇𑌲𑌾𑌭𑌙𑍍𑌗𑌪𑍍𑌰𑌶𑌮𑌨𑌫𑌲𑌂 𑌵𑍈𑌦𑌿𑌕𑌂 𑌧𑌰𑍍𑌮𑌸𑍇𑌤𑍁𑌮𑍍 ॥௰௯॥

𑌸𑍇𑌵𑍇 𑌦𑍇𑌵𑌿 𑌤𑍍𑌰𑌿𑌦𑌶𑌮𑌹𑌿𑌲𑌾𑌮𑍌𑌲𑌿𑌮𑌾𑌲𑌾𑌰𑍍𑌚𑌿𑌤𑌂 𑌤𑍇 𑌸𑌿𑌦𑍍𑌧𑌿𑌕𑍍𑌷𑍇𑌤𑍍𑌰𑌂 𑌶𑌮𑌿𑌤𑌵𑌿𑌪𑌦𑌾𑌂 𑌸𑌮𑍍𑌪𑌦𑌾𑌂 𑌪𑌾𑌦𑌪𑌦𑍍𑌮𑌮𑍍 ।
𑌯𑌸𑍍𑌮𑌿𑌨𑍍𑌨𑍀𑌷𑌨𑍍𑌨𑌮𑌿𑌤𑌶𑌿𑌰𑌸𑍋 𑌯𑌾𑌪𑌯𑌿𑌤𑍍𑌵𑌾 𑌶𑌰𑍀𑌰𑌂 𑌵𑌰𑍍𑌤𑌿𑌷𑍍𑌯𑌨𑍍𑌤𑍇 𑌵𑌿𑌤𑌮𑌸𑌿 𑌪𑌦𑍇 𑌵𑌾𑌸𑍁𑌦𑍇𑌵𑌸𑍍𑌯 𑌧𑌨𑍍𑌯𑌾𑌃 ॥௨௰॥

𑌸𑌾𑌨𑍁𑌪𑍍𑌰𑌾𑌸𑌪𑍍𑌰𑌕𑌟𑌿𑌤𑌦𑌯𑍈𑌃 𑌸𑌾𑌨𑍍𑌦𑍍𑌰𑌵𑌾𑌤𑍍𑌸𑌲𑍍𑌯𑌦𑌿𑌗𑍍𑌧𑍈𑌰𑌮𑍍𑌬 𑌸𑍍𑌨𑌿𑌗𑍍𑌧𑍈𑌰𑌮𑍃𑌤𑌲𑌹𑌰𑍀𑌲𑌬𑍍𑌧𑌸𑌬𑍍𑌰𑌹𑍍𑌮𑌚𑌰𑍍𑌯𑍈𑌃 ।
𑌘𑌰𑍍𑌮𑍇 𑌤𑌾𑌪𑌤𑍍𑌰𑌯𑌵𑌿𑌰𑌚𑌿𑌤𑍇 𑌗𑌾𑌢𑌤𑌪𑍍𑌤𑌂 𑌕𑍍𑌷𑌣𑌂 𑌮𑌾𑌮𑌾𑌕𑌿𑌞𑍍𑌚𑌨𑍍𑌯𑌗𑍍𑌲𑌪𑌿𑌤𑌮𑌨𑌘𑍈𑌰𑌾𑌰𑍍𑌦𑍍𑌰𑌯𑍇𑌥𑌾𑌃 𑌕𑌟𑌾𑌕𑍍𑌷𑍈𑌃 ॥௨௰௧॥

𑌸𑌮𑍍𑌪𑌦𑍍𑌯𑌨𑍍𑌤𑍇 𑌭𑌵𑌭𑌯𑌤𑌮𑍀𑌭𑌾𑌨𑌵𑌸𑍍𑌤𑍍𑌵𑌤𑍍𑌪𑍍𑌰𑌸𑌾𑌦𑌾𑌦𑍍𑌭𑌾𑌵𑌾𑌃 𑌸𑌰𑍍𑌵𑍇 𑌭𑌗𑌵𑌤𑌿 𑌹𑌰𑍌 𑌭𑌕𑍍𑌤𑌿𑌮𑍁𑌦𑍍𑌵𑍇𑌲𑌯𑌨𑍍𑌤𑌃 ।
𑌯𑌾𑌚𑍇 𑌕𑌿𑌂 𑌤𑍍𑌵𑌾𑌮𑌹𑌮𑌿𑌹 𑌯𑌤𑌃 𑌶𑍀𑌤𑌲𑍋𑌦𑌾𑌰𑌶𑍀𑌲𑌾 𑌭𑍂𑌯𑍋 𑌭𑍂𑌯𑍋 𑌦𑌿𑌶𑌸𑌿 𑌮𑌹𑌤𑌾𑌂 𑌮𑌙𑍍𑌗𑌲𑌾𑌨𑌾𑌂 𑌪𑍍𑌰𑌬𑌨𑍍𑌧𑌾𑌨𑍍 ॥௨௰௨॥

𑌮𑌾𑌤𑌾 𑌦𑍇𑌵𑌿 𑌤𑍍𑌵𑌮𑌸𑌿 𑌭𑌗𑌵𑌾𑌨𑍍𑌵𑌾𑌸𑍁𑌦𑍇𑌵𑌃 𑌪𑌿𑌤𑌾 𑌮𑍇 𑌜𑌾𑌤𑌃 𑌸𑍋𑌹𑌂 𑌜𑌨𑌨𑌿 𑌯𑍁𑌵𑌯𑍋𑌰𑍇𑌕𑌲𑌕𑍍𑌷𑍍𑌯𑌂 𑌦𑌯𑌾𑌯𑌾𑌃 ।
𑌦𑌤𑍍𑌤𑍋 𑌯𑍁𑌷𑍍𑌮𑌤𑍍𑌪𑌰𑌿𑌜𑌨𑌤𑌯𑌾 𑌦𑍇𑌶𑌿𑌕𑍈𑌰𑌪𑍍𑌯𑌤𑌸𑍍𑌤𑍍𑌵𑌂 𑌕𑌿𑌂 𑌤𑍇 𑌭𑍂𑌯𑌃 𑌪𑍍𑌰𑌿𑌯𑌮𑌿𑌤𑌿 𑌕𑌿𑌲 𑌸𑍍𑌮𑍇𑌰𑌵𑌕𑍍𑌰𑌾 𑌵𑌿𑌭𑌾𑌸𑌿 ॥௨௰௩॥

𑌕𑌲𑍍𑌯𑌾𑌣𑌾𑌨𑌾𑌮𑌵𑌿𑌕𑌲𑌨𑌿𑌧𑌿𑌃 𑌕𑌾𑌪𑌿 𑌕𑌾𑌰𑍁𑌣𑍍𑌯𑌸𑍀𑌮𑌾 𑌨𑌿𑌤𑍍𑌯𑌾𑌮𑍋𑌦𑌾 𑌨𑌿𑌗𑌮𑌵𑌚𑌸𑌾𑌂 𑌮𑍌𑌲𑌿𑌮𑌨𑍍𑌦𑌾𑌰𑌮𑌾𑌲𑌾 ।
𑌸𑌮𑍍𑌪𑌦𑍍𑌦𑌿𑌵𑍍𑌯𑌾 𑌮𑌧𑍁𑌵𑌿𑌜𑌯𑌿𑌨𑌃 𑌸𑌨𑍍𑌨𑌿𑌧𑌤𑍍𑌤𑌾𑌂 𑌸𑌦𑌾 𑌮𑍇 𑌸𑍈𑌷𑌾 𑌦𑍇𑌵𑍀 𑌸𑌕𑌲𑌭𑍁𑌵𑌨𑌪𑍍𑌰𑌾𑌰𑍍𑌥𑌨𑌾𑌕𑌾𑌮𑌧𑍇𑌨𑍁𑌃 ॥௨௰௪॥

𑌉𑌪𑌚𑌿𑌤𑌗𑍁𑌰𑍁𑌭𑌕𑍍𑌤𑍇𑌰𑍁𑌤𑍍𑌥𑌿𑌤𑌂 𑌵𑍇𑌙𑍍𑌕𑌟𑍇𑌶𑌾𑌤𑍍𑌕𑌲𑌿𑌕𑌲𑍁𑌷𑌨𑌿𑌵𑍃𑌤𑍍𑌤𑍍𑌯𑍈 𑌕𑌲𑍍𑌪𑍍𑌯𑌮𑌾𑌨𑌂 𑌪𑍍𑌰𑌜𑌾𑌨𑌾𑌮𑍍 ।
𑌸𑌰𑌸𑌿𑌜𑌨𑌿𑌲𑌯𑌾𑌯𑌾𑌃 𑌸𑍍𑌤𑍋𑌤𑍍𑌰𑌮𑍇𑌤𑌤𑍍𑌪𑌠𑌨𑍍𑌤𑌃 𑌸𑌕𑌲𑌕𑍁𑌶𑌲𑌸𑍀𑌮𑌾 𑌸𑌾𑌰𑍍𑌵𑌭𑍌𑌮𑌾 𑌭𑌵𑌨𑍍𑌤𑌿 ॥௨௰௫॥

𑌕𑌵𑌿𑌤𑌾𑌰𑍍𑌕𑌿𑌕𑌸𑌿𑌂𑌹𑌾𑌯 𑌕𑌲𑍍𑌯𑌾𑌣𑌗𑍁𑌣𑌶𑌾𑌲𑌿𑌨𑍇 ।
𑌶𑍍𑌰𑍀𑌮𑌤𑍇 𑌵𑍇𑌙𑍍𑌕𑌟𑍇𑌶𑌾𑌯 𑌵𑍇𑌦𑌾𑌨𑍍𑌤𑌗𑍁𑌰𑌵𑍇 𑌨𑌮𑌃 ॥
